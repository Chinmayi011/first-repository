/*
 * Complete the 'roadsAndLibraries' function below.
 *
 * The function is expected to return a LONG_INTEGER.
 * The function accepts following parameters:
 *  1. INTEGER n
 *  2. INTEGER c_lib
 *  3. INTEGER c_road
 *  4. 2D_INTEGER_ARRAY cities
 */
typedef struct {
    int *data;
    int size;
    int capacity;
} Vector;

static void vec_init(Vector *v) {
    v->size = 0;
    v->capacity = 2;
    v->data = malloc(v->capacity * sizeof(int));
}

static void vec_push(Vector *v, int val) {
    if (v->size == v->capacity) {
        v->capacity *= 2;
        v->data = realloc(v->data, v->capacity * sizeof(int));
    }
    v->data[v->size++] = val;
}

static void vec_free(Vector *v) {
    free(v->data);
}
long roadsAndLibraries(int n, int c_lib, int c_road, int cities_rows, int cities_columns, int** cities) {
     if (c_lib <= c_road) {
        return (long)n * c_lib;
    }

    // Build adjacency list
    Vector *adj = malloc((n + 1) * sizeof(Vector));
    for (int i = 1; i <= n; i++) vec_init(&adj[i]);

    for (int i = 0; i < cities_rows; i++) {
        int u = cities[i][0];
        int v = cities[i][1];
        vec_push(&adj[u], v);
        vec_push(&adj[v], u);
    }

    bool *visited = calloc(n + 1, sizeof(bool));
    long total_cost = 0;

    // BFS queue (reused)
    int *queue = malloc(n * sizeof(int));

    for (int i = 1; i <= n; i++) {
        if (!visited[i]) {
            // new connected component
            long cities_in_comp = 0;

            int front = 0, back = 0;
            queue[back++] = i;
            visited[i] = true;

            while (front < back) {
                int node = queue[front++];
                cities_in_comp++;

                for (int k = 0; k < adj[node].size; k++) {
                    int neighbor = adj[node].data[k];
                    if (!visited[neighbor]) {
                        visited[neighbor] = true;
                        queue[back++] = neighbor;
                    }
                }
            }

            // cost = 1 library + (cities_in_comp - 1) roads
            total_cost += c_lib + (cities_in_comp - 1) * c_road;
        }
    }

    // Free memory
    for (int i = 1; i <= n; i++) vec_free(&adj[i]);
    free(adj);
    free(queue);
    free(visited);

    return total_cost;

}

char* readline() {
    size_t alloc_length = 1024;
    size_t data_length = 0;

    char* data = malloc(alloc_length);

    while (true) {
        char* cursor = data + data_length;
        char* line = fgets(cursor, alloc_length - data_length, stdin);

        if (!line) {
            break;
        }

        data_length += strlen(cursor);

        if (data_length < alloc_length - 1 || data[data_length - 1] == '\n') {
            break;
        }

        alloc_length <<= 1;

        data = realloc(data, alloc_length);

        if (!data) {
            data = '\0';

            break;
        }
    }

    if (data[data_length - 1] == '\n') {
        data[data_length - 1] = '\0';

        data = realloc(data, data_length);

        if (!data) {
            data = '\0';
        }
    } else {
        data = realloc(data, data_length + 1);

        if (!data) {
            data = '\0';
        } else {
            data[data_length] = '\0';
        }
    }

    return data;
}

char* ltrim(char* str) {
    if (!str) {
        return '\0';
    }

    if (!*str) {
        return str;
    }

    while (*str != '\0' && isspace(*str)) {
        str++;
    }

    return str;
}

char* rtrim(char* str) {
    if (!str) {
        return '\0';
    }

    if (!*str) {
        return str;
    }

    char* end = str + strlen(str) - 1;

    while (end >= str && isspace(*end)) {
        end--;
    }

    *(end + 1) = '\0';

    return str;
}

char** split_string(char* str) {
    char** splits = NULL;
    char* token = strtok(str, " ");

    int spaces = 0;

    while (token) {
        splits = realloc(splits, sizeof(char*) * ++spaces);

        if (!splits) {
            return splits;
        }

        splits[spaces - 1] = token;

        token = strtok(NULL, " ");
    }

    return splits;
}

int parse_int(char* str) {
    char* endptr;
    int value = strtol(str, &endptr, 10);

    if (endptr == str || *endptr != '\0') {
        exit(EXIT_FAILURE);
    }

    return value;
}
