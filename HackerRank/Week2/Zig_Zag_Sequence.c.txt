#include <stdio.h>
#include <stdlib.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}


int cmpfunc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}


void findZigZagSequence(int a[], int n) {
   int mid = (n - 1) / 2;

    // Step 1: Sort the array
    qsort(a, n, sizeof(int), cmpfunc);

    // Step 2: Swap middle and last element
    swap(&a[mid], &a[n - 1]);

    // Step 3: Reverse the segment after mid up to n-2
    int st = mid + 1, ed = n - 2;
    while (st < ed) {
        swap(&a[st], &a[ed]);
        st++;
        ed--;
    }

    // Print result with single spaces
    for (int i = 0; i < n; i++) {
        printf("%d", a[i]);
        if (i < n - 1) printf(" ");
    }
    printf("\n");
    }

    
    

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int a[n];
        for (int i = 0; i < n; i++)
            scanf("%d", &a[i]);
        findZigZagSequence(a, n);
    }
    return 0;
}
