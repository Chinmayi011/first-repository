#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

#define MAX 100000

typedef struct {
    long long data[MAX];
    int top;
} Stack;

void push(Stack *s, long long val) {
    s->data[++(s->top)] = val;
}

long long pop(Stack *s) {
    return s->data[(s->top)--];
}

long long peek(Stack *s) {
    return s->data[s->top];
}

int isEmpty(Stack *s) {
    return s->top == -1;
}

int main() {
    int q;
    scanf("%d", &q);

    Stack stack_in, stack_out;
    stack_in.top = -1;
    stack_out.top = -1;

    for(int i = 0; i < q; i++) {
        int type;
        scanf("%d", &type);

        if (type == 1) {
            long long x;
            scanf("%lld", &x);
            push(&stack_in, x);
        }
        else if (type == 2) {
            if (isEmpty(&stack_out)) {
                while (!isEmpty(&stack_in))
                    push(&stack_out, pop(&stack_in));
            }
            pop(&stack_out);
        }
        else if (type == 3) {
            if (isEmpty(&stack_out)) {
                while (!isEmpty(&stack_in))
                    push(&stack_out, pop(&stack_in));
            }
            printf("%lld\n", peek(&stack_out));
        }
    }


    /* Enter your code here. Read input from STDIN. Print output to STDOUT */    
    return 0;
}
