#include <stdio.h>
#include<string.h>

int main(){
    int n,q,i,j,count;
    scanf("%d",&n);
    char strings[n][21];
    for (i=0;i<n;i++)
        scanf("%s", strings[i]);
    
    scanf("%d",&q);
    char queries[q][21];
    for(i=0;i<q;i++)
        scanf("%s",queries[i]);
    
    for(i=0;i<q;i++){
        count=0;
        for(j=0;j<n;j++)
            if(strcmp(queries[i], strings[j])==0)
                count++;
        printf("%d\n", count);
        
    }
    return 0;
}
