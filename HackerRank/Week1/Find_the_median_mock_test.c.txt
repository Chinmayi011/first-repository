
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

/*
 * Complete the 'findMedian' function below.
 * The function is expected to return an INTEGER.
 * The function accepts:
 *   1. INTEGER arr_count -> number of elements
 *   2. INTEGER ARRAY arr -> input numbers
 */

int compare(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int findMedian(int arr_count, int arr[]) {
    // Sort the array
    qsort(arr, arr_count, sizeof(int), compare);
    
    // Since n is odd, median is the middle element
    int mid = arr_count / 2;
    return arr[mid];
}

int main() {
    int n;
    scanf("%d", &n);
    assert(n > 0 && (n % 2 == 1));  // check odd size
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    int result = findMedian(n, arr);
    printf("%d\n", result);
    
    return 0;
}
